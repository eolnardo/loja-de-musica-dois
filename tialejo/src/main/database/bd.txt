CREATE TABLE Produto (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    avaliacao VARCHAR(10),
    descricao VARCHAR(255),
    preco DECIMAL(10, 2),
    qtdEstoque INT,
    status BOOLEAN,
    image VARCHAR(255)
);


CREATE TABLE Usuario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(50),
    email VARCHAR(100),
    grupo VARCHAR(100),
    cpf VARCHAR(14),
    senha VARCHAR(50),
    confirmaSenha VARCHAR(50),
    status boolean
);

CREATE TABLE Cliente (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    telefone VARCHAR(20) NOT NULL,
    genero VARCHAR(10),
    cpf VARCHAR(14) UNIQUE NOT NULL,
    senha VARCHAR(100) NOT NULL,
    confirmaSenha VARCHAR(100) NOT NULL,
    enderecoFaturamento VARCHAR(255) NOT NULL,
    enderecoEntrega VARCHAR(255) NOT NULL,
    dataNascimento DATE,
    CONSTRAINT nome_2_palavras CHECK (nome REGEXP '^[a-zA-Z]+ [a-zA-Z]+$'),
    CONSTRAINT senha_unique UNIQUE (senha)
);
